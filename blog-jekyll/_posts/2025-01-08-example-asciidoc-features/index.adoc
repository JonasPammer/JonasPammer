.Python
[source,python,linenums]
----
def fibonacci(n):
    """Generate Fibonacci sequence up to n terms."""
    a, b = 0, 1
    result = []
    for _ in range(n):
        result.append(a)
        a, b = b, a + b
    return result

# Example usage
print(fibonacci(10))
----

.JavaScript
[source,javascript]
----
// FlexSearch configuration
const index = new FlexSearch.Document({
  document: {
    id: "url",

  },

});

// Add documents
data.forEach(doc => index.add(doc));
----

.YAML
[source,yaml]
----
name: Deploy Blog
on:
  push:
    branches: [master]
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Build
        run: hugo --minify
----
